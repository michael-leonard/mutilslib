00224 
00225 implicit none
00226 ! Dummies
00227 real(mrk),intent(in)::x(:)
00228 real(mrk),intent(out)::mean
00229 real(mrk),intent(out),optional::var
00230 character(*),intent(in)::method
00231 integer(mik),intent(out)::err
00232 character(*),intent(out)::message
00233 ! Locals
00234 real(mrk),allocatable::s(:)
00235 integer(mik)::ns
00236 real(mrk)::nsr
00237 ! Start procedure here
00238 err=0; message="getmeanvar_s/ok"; ns=size(x); nsr=real(ns,mrk)
00239 mean=sum(x(:))/nsr ! get approximate expectation of x first
00240 if(present(var))then ! get approximate variance if requested
00241   if(ns==1)then
00242     err=-100; var=hugeRe; message="w-getmeanvar_s/singleSample"; return
00243   endif
00244   selectcase(method)
00245   case("f")     ! fast scheme:    var[x]=E[x2]-E[x].E[x]
00246     var=sum(x(:)**2)/real(ns-1,mrk)-mean**2
00247   case("c")     ! classic scheme: var[x]=E[(x-mean)^2]
00248     var=sum((x(:)-mean)**2)/real(ns-1,mrk)
00249   case default  ! 2-pass formula to minimise round-off errors
00250     allocate(s(ns),stat=err)
00251     if(err/=0)then
00252       err=10;message="f-getmeanvar_s/allocError";return
00253     endif
00254     s(:)=x(:)-mean ! compute deviations for each point
00255     var=(sum(s(:)**2)-sum(s(:))**2/nsr)/real(ns-1,mrk)
00256     deallocate(s,stat=err)
00257     if(err/=0)then
00258       err=20;message="f-getmeanvar_s/deAllocError";return
00259     endif
00260   endselect
00261 endif
00262 ! End procedure here
00263 endsubroutine getmeanvar_s
